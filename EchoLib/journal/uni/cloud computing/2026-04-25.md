from contextlib import asynccontextmanager

@asynccontextmanager
async def lifespan(app: FastAPI):
    # This runs on startup
    db = next(get_db())
    try:
        print("Recalculating all movie averages...")
        # Efficient bulk update using SQLAlchemy
        avg_query = (
            db.query(RatingTable.movie_id, func.avg(RatingTable.rating).label('avg_res'))
            .filter(RatingTable.is_quarantined == False)
            .group_by(RatingTable.movie_id)
            .all()
        )
        
        for movie_id, avg_val in avg_query:
            db.query(Movie).filter(Movie.movie_id == movie_id).update({"avg_rating": float(avg_val)})
        
        db.commit()
        print("Averages populated successfully.")
    except Exception as e:
        print(f"Startup calculation failed: {e}")
        db.rollback()
    finally:
        db.close()
    yield
