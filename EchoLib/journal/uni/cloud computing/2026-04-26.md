@app.get("/recommendations/{user_id}", response_model=List[Recommendation])
async def get_recommendations(
    user_id: int,
    db: Session = Depends(get_db),
    user_client: UserGrpcClient = Depends(get_user_client),
    movie_client: MovieGrpcClient = Depends(get_movie_client),
    review: ReviewGrpcClient = Depends(get_review_client),
):
    # 1. Validation
    user_exists = await user_client.validate_user(user_id)
    if not user_exists:
        raise HTTPException(status_code=404, detail="User does not exist")
    
    # 2. Get User Ratings
    ratings = await review.get_user_ratings(user_id)
    ratings = [r for r in ratings if r.rating >= 3.0]
    rated_movie_ids = [r.movie_id for r in ratings]
    
    # 3. Get Reference Movies (Local table)
    ref_entries = db.query(UserReferenceMovieTable).filter(UserReferenceMovieTable.user_id == user_id).all()
    ref_movie_ids = [ref.movie_id for ref in ref_entries]

    # 4. FETCH ALL MOVIE DATA VIA GRPC (Replacing direct Movie table queries)
    all_needed_ids = list(set(rated_movie_ids + ref_movie_ids))
    # Use the batch method already in your proto
    grpc_movies = await movie_client.get_movies_batch(all_needed_ids) 
    movie_lookup = {m.movie_id: m for m in grpc_movies}

    # 5. Calculate Genre Preferences
    genre_pref = {}
    
    # Process Ratings
    for r in ratings:
        m = movie_lookup.get(r.movie_id)
        if m:
            for g_name in m.genres: # m.genres is now a List[str] from gRPC
                genre_pref[g_name] = genre_pref.get(g_name, 0) + float(r.rating)

    # Process Reference Movies
    for r_id in ref_movie_ids:
        m = movie_lookup.get(r_id)
        if m:
            for g_name in m.genres:
                genre_pref[g_name] = genre_pref.get(g_name, 0) + 1.5

    # Process User Preference "Likes/Dislikes" (Requires Genre Names)
    user_prefs = db.query(UserPreferenceTable).filter(UserPreferenceTable.user_id == user_id).all()
    for p in user_prefs:
        # Optimization: Use the GetGenres gRPC or a local genre cache 
        # For now, keeping your local Genre table query if that table exists in this service
        genre = db.query(Genre).filter(Genre.genre_id == p.genre_id).first()
        if genre:
            weight = 2.0 if p.preference_type == "like" else -3.0
            genre_pref[genre.name] = genre_pref.get(genre.name, 0) + weight

    if not genre_pref:
        # Refactor this to use movie_client.search_movies or similar
        return await get_top_rated_recommendations(movie_client)
    
    # 6. Get Candidates via gRPC
    exclude_ids = set(all_needed_ids)
    # Fetch a pool of candidates (e.g., top 100 or recent)
    candidates = await movie_client.search_movies(limit=100) 

    scored_movies = []
    for m in candidates:
        if m.movie_id in exclude_ids:
            continue
        score = sum(genre_pref.get(g_name, 0.0) for g_name in m.genres)
        if score > 0:
            scored_movies.append((score, m))

    scored_movies.sort(key=lambda x: x[0], reverse=True)
    return [Recommendation(movie_id=m.movie_id, title=m.movie_title) for _, m in scored_movies[:5]]

@app.get("/movies/{movie_id}/ratings", response_model=List[Rating])
async def get_movie_ratings(
    movie_id: int = Path(description="The ID of the movie to retrieve ratings for"),
    user_id: Optional[int] = Query(None, description="Filter ratings by user ID"),
    min_rating: Optional[float] = Query(None, ge=1.0, le=5.0, description="Filter ratings with a minimum rating value"),
    max_rating: Optional[float] = Query(None, ge=1.0, le=5.0, description="Filter ratings with a maximum rating value"),
    tag: Optional[str] = Query(None, description="Filter ratings by tag"),
    # New parameters for pagination to prevent loading thousands of records at once
    limit: int = Query(10, ge=1, le=100, description="Number of results to return"),
    offset: int = Query(0, ge=0, description="Number of results to skip"),
    movie_client: MovieGrpcClient = Depends(get_movie_client),
    db: Session = Depends(get_db)
):
    try:
        # gRPC call introduces network latency; consider caching these results if movies are frequently accessed
        movie_exists = await movie_client.validate_movie(movie_id)
        if not movie_exists:
            raise HTTPException(status_code=404, detail="Movie does not exist")
        
        # Start building the query
        query = db.query(RatingTable).filter(
            RatingTable.movie_id == movie_id, 
            RatingTable.is_quarantined == False
        )

        # Apply filters
        if user_id is not None:
            query = query.filter(RatingTable.user_id == user_id)
        if min_rating is not None:
            query = query.filter(RatingTable.rating >= min_rating)
        if max_rating is not None:
            query = query.filter(RatingTable.rating <= max_rating)
        if tag is not None:
            # PERFORMANCE NOTE: Overlap searches on unindexed columns are slow
            query = query.filter(RatingTable.tag.overlap(tag))
        
        # PERFORMANCE FIX 1: Removed 'query.count()'. 
        # Executing count() and all() separately doubles the DB execution time.
        # PERFORMANCE FIX 2: Added limit and offset for pagination.
        results = query.offset(offset).limit(limit).all()
        
        if not results:
            raise HTTPException(status_code=404, detail="No ratings found")
        
        return results
        
    except HTTPException:
        raise
    except Exception as e:
        print(f"Error retrieving ratings: {e}")
        raise HTTPException(status_code=500, detail="Error retrieving ratings")

@app.get("/ratings", response_model=List[Rating])
async def get_ratings(
    user_id: Optional[int] = Query(None),
    movie_id: Optional[int] = Query(None),
    min_rating: Optional[float] = Query(None, ge=1.0, le=5.0),
    max_rating: Optional[float] = Query(None, ge=1.0, le=5.0),
    limit: int = Query(20, ge=1, le=100), # Performance fix: Pagination
    offset: int = Query(0, ge=0),
    db: Session = Depends(get_db)
):
    query = db.query(RatingTable).filter(RatingTable.is_quarantined == False)
    
    if user_id: query = query.filter(RatingTable.user_id == user_id)
    if movie_id: query = query.filter(RatingTable.movie_id == movie_id)
    if min_rating: query = query.filter(RatingTable.rating >= min_rating)
    if max_rating: query = query.filter(RatingTable.rating <= max_rating)
    
    # Apply pagination
    return query.offset(offset).limit(limit).all()

@app.get("/movies/{movie_id}/ratings", response_model=List[Rating])
async def get_movie_ratings(
    movie_id: int = Path(...),
    limit: int = Query(10, ge=1, le=50),
    offset: int = Query(0, ge=0),
    movie_client: MovieGrpcClient = Depends(get_movie_client),
    db: Session = Depends(get_db)
):
    # 1. Validate via gRPC
    if not await movie_client.validate_movie(movie_id):
        raise HTTPException(status_code=404, detail="Movie does not exist")
    
    # 2. Fetch results directly. Avoid redundant count()
    results = db.query(RatingTable).filter(
        RatingTable.movie_id == movie_id,
        RatingTable.is_quarantined == False
    ).offset(offset).limit(limit).all()
    
    if not results:
        raise HTTPException(status_code=404, detail="No ratings found")
    
    return results
