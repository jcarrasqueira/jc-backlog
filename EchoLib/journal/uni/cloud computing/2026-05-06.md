gcloud iam service-accounts create review-intelligence-sa --display-name="Review Intelligence SA"

    --member="serviceAccount:${SA_EMAIL}" \

    --member="serviceAccount:${SA_EMAIL}" \
    --role="roles/aiplatform.serviceAgent"
